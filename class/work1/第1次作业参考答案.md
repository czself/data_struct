#### 1.登录

要求有：

- 从函数运行处开始登录（main函数的参数），第一个参数为用户名，第二参数为密码，其余参数一律不统计在内。
- 如果main函数参数没有输入，则要求在内部设置scanf函数并提示用户登录。
- 直至登录成功为止

```c
#include<stdio.h>
#include<string.h>

#define MAX 32
#define USERNAME "wangzhendong"
#define PASSWORD "12345678"

int main(int arg, char** args){
   	/**
   	char** args 是一个字符串数组。
   	arg指的是这个数值的数量
   	*/
    
	char username[MAX] = {'\0'};
	char password[MAX] = {'\0'};
	if(arg >= 2){
		char *n = args[1];
		strncpy(username,n,strlen(n));   
        //将字符串数组的第二项（index==1） copy到 username数组中。
	}
	if(arg >= 3){
		char *p = args[2];
		strncpy(password,p,strlen(p));
	}
	
	while(1){
		if(strlen(username) > 0 && strlen(password) > 0 ){
			if(strncmp(username,USERNAME,MAX) == 0){
				if(strncmp(password,PASSWORD,MAX) == 0){
					printf("欢迎进入本系统\n");
					break;
				}else{
					memset(password, '\0', MAX);
					printf("密码错误，请重新输入。\n");
				}
			}
			else{
				memset(username, '\0', MAX);
				memset(password, '\0', MAX);
				printf("您输入的用户不存在，请重新输入。\n");
			}
		}
		
		if(strlen(username) == 0){
			printf("请输入用户名：");
			scanf("%s",username);
		}
		if(strlen(password) == 0){
			printf("请输入密码：");
			scanf("%s",password); 
		}
	} 
	return 0;
} 
```



#### 2.编写一个函数，名字为isNumber`，参数为一个字符串，判断它是不是数字——不考虑科学计数法。

```c
#include <stdio.h>
#include <stdbool.h>
#include <string.h>

bool isNumber(char* str){
	if(str == NULL) return false;
	int len = strlen(str);
	int flag = 0;
	int dot = 0;
	
    /***
    在编程时，应多做排除法，否则if就会嵌套严重
    */
    
    /*首先排除的是+和-*/
	if (str[0] == '+' || str[0] == '-'){
		flag = 1;
	}
    
	int i = flag;  
    // ↑ 如果首字符是+或者-，则flag为1，下面的for就从1开始；因为 str[0]已经遍历了
    //   如果首字符不是+或者-，则flag为0，从str[0]开始遍历
	for(;i<len;i++){
		char c = str[i];
        
        // 还是排除法，先把 小数点 找出来。
		if(c == '.'){
			dot++;
			if(dot == 2 ){
                // 如果有两个小数点，表示不是数字，直接返回。
				return false;
			}
			continue;
            // 如果是小数点，不要再判断它是不是数字了。
		}
		// ↓ 除了小数点外，其他的都是数字字符。不是数字字符，就返回
		if(c > '9' || c < '0'){
			return false;	
		}
	}
	return true;
}

int main(){
	char* p = "+.1234";
	printf("%d\n",isNumber(p));
	return 0;
} 
```

#### 6.编写一个函数，名字为`number2String`，参数任意类型的数字，将其换成字符串。

如输入 1234.1，输出“1234.1”

```c
//这是最难的一题
#define MAX 64
/*这里面我们直接用char c[],且不用传递长度，因为MAX已经被宏定义了*/
void number2String(float num, char c[]){
    
	int integer = (int) num ;   // 获得整数部分
	float flo = num - integer;  // 获得小数部分
	int i = 0;
    
    /**↓↓↓获得整数的字符*/
	char cInt[MAX] = {'\0'}; 
    /**这里面不能全都赋值为0，因为会出现问题，这是一个同学提出来的，我们还是赋值为-1，或者非数字字符*/
	memset(cInt,'a' MAX);
	while(1){
		cInt[i] =(char) (integer % 10 + '0');
		integer = (int)integer / 10;
		i++;
		if(integer == 0){
			break;
		}		
	}
    
    
    /**↓↓↓获得小数的字符*/
	char cFloat[MAX] = {'\0'};
    memset(cFloat,'a',MAX);
	i = 0;
	while(1){
		int j = (int)flo*10;
		cFloat[i] =(char) (j+'0');
		flo = flo * 10 - j;
		i++;
		if(flo == 0.0){
			break;
		}
	}
    
    /**↓↓↓将cFloat 和cInt拼在一起*/
    //整数部分
	int k = 0;
	for(i = MAX-1;i>=0;i--){
		if(cInt[i] == 'a' ){   
            //在2班上课的时候，这里面用了0，一个同学指出了问题，因为假设整数102，在这个地方会出现问题
            //所以我们用了a——一个非数字字符。
			continue;
		}
		c[k] = cInt[i];
		k++;	
	}
     //小数点
	c[k] == '.' ;
    k++;
    //小数部分
	for(i = 0; i<MAX; i++){
        if(cFloat[i] == 'a'){
            c[k] = 0;
            break;
        }
       	c[k] = cFloat[i];
        k++;
    }
}


int main(){
	char mc[MAX] = {'\0'};
	
	float f = 123.123;
	number2String(f,mc);
}
```

#### 7.编写一个函数，名字为stringCat,参数为两个字符串，返回的是这两个字符串相加后的新字符串。

如”123“，”ABC“，得到”123ABC“

```c
char* stringCat(char *a,char* b){
	if(a == NULL || b == NULL) 
        return NULL;
	
	int len =  (strlen(a)+strlen(b)) + 1;
    
    /*
    在堆上分配内存
    */
	char *c = (char*) malloc( sizeof(char) *  len);
    
    /**将内存初始化为0*/
	memset(c,'\0',len);
	
	int i = 0;
	for(;i<strlen(a);i++){
		c[i] = a[i];
	}
	
	for(i= 0 ; i<strlen(b);i++){
		c[i+strlen(a)] = b[i];
	}	
	return c;
}

int main(){
	char *a = "abc";
	char *b = "123";
	char *p = stringCat(a,b);
	a = p;
	
	free(p);
}
```

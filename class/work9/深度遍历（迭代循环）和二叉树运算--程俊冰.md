## 深度遍历（循环）

```c++
#include <unordered_map>
#include <vector>
#include <string>
#include <filesystem>
#include <iostream>
#include <stack>
#include <variant>

using namespace std;
namespace fs = std::filesystem;
using VAR = variant<int, string, bool>;

// 不知道怎么输出variant的数据，所以用AI重载了<<
std::ostream& operator<<(std::ostream& os,
                         const std::variant<int, std::string, bool>& v) {
    std::visit([&os](const auto& value) {
        os << value;
    }, v);
    return os;
}

struct Dir {
    std::string path;
    int count = 0;
    int id = 0;
};
// 这里用结构体中的count保存了目录已经遍历过的文件个数，顺便也把文件的级数（id）也加到了结构体中


// 我使用栈来储存没有遍历结束的目录，并储存每个目录遍历的次数，用于跳过下一次遍历出现的重复数据，当栈顶目录遍历结束后出栈，进行下一个栈顶的遍历，不过我这个方法在性能上还是很差的
vector<unordered_map<string, VAR>> depthFirstSearch(const Dir& root, unordered_map<string, VAR> dirMap) {
    stack<Dir> dirStack;
    dirStack.push(root);
    vector<unordered_map<string, VAR>> dirs;

    dirs.push_back(dirMap);

    while (!dirStack.empty()) {
        int num = dirStack.top().count;
        bool finish = true;
        for (auto &item : fs::directory_iterator(dirStack.top().path)) {
            if (num > 0) {
                num--;
                continue;
            }

            dirMap["path"] = item.path().string();
            dirMap["name"] = item.path().filename().string();
            dirMap["id"] = dirStack.top().id + 1;
            dirs.push_back(dirMap);
            dirStack.top().count++;

            if (fs::is_directory(item)) {
                Dir dir;
                dir.path = item.path().string();
                dir.id = dirStack.top().id + 1;
                dirStack.push(dir);
                finish = false;
                break;
            }
        }
        if (finish) {
            dirStack.pop();
        }
    }

    return dirs;
}

int main() {

    Dir root;
    root.path = "D:/c++/textDir";

    unordered_map<string, VAR> dirMap;
    dirMap["path"] = "D:/c++/textDir";
    dirMap["name"] = "textDir";
    dirMap["id"] = 0;
    vector<unordered_map<string, VAR>> dirs = depthFirstSearch(root, dirMap);

    for (auto &item : dirs) {
        cout << item["id"] << "   " <<  item["path"] << "   " << item["name"] << endl;
    }


    return 0;

}
```

## 二叉树运算

```c++
#include <vector>
#include <string>
#include <iostream>

using namespace std;

struct BinaryTree {
    int data;
    char op;
    BinaryTree* left;
    BinaryTree* right;
};

BinaryTree* expressionToBinaryTree(const string& expression) {
    BinaryTree* root = nullptr;
    for (auto item : expression) {
        BinaryTree* treeNode = new BinaryTree();
        if (item == '+' || item == '-' || item == '*' || item == '/') {
            treeNode -> op = item;
            treeNode -> data = -1;
        } else {
            treeNode -> data = item - '0';
            treeNode -> op = '#';
        }

        if (root == nullptr) {
             root = treeNode;
             continue;
         }

        if (treeNode -> op == '+' || treeNode -> op == '-') {
            treeNode -> left = root;
            root = treeNode;
            continue;
        }

        if (treeNode -> op == '*' || treeNode -> op == '/') {
            if (root -> op == '+' || root -> op == '-') {
                treeNode -> left = root -> right;
                root -> right = treeNode;
                continue;
            }

            treeNode -> left = root;
            root = treeNode;
            continue;
        }

        if (treeNode -> op == '#') {
            BinaryTree* p = root;
            while (p -> right != nullptr) {
                p = p -> right;
            }
            p -> right = treeNode;
            continue;
        }

    }

    return root;
}


bool evaluateBinaryTree(BinaryTree* root) {
    if (root == nullptr) return false;

    if (evaluateBinaryTree(root->left) && evaluateBinaryTree(root->right)) {
        if (root -> op == '+') {
            root -> data = root -> left -> data + root -> right -> data;
        } else if (root -> op == '-') {
            root -> data = root -> left -> data - root -> right -> data;
        } else if (root -> op == '*') {
            root -> data = root -> left -> data * root -> right -> data;
        } else if (root -> op == '/') {
            root -> data = root -> left -> data / root -> right -> data;
        }

    }

    return true;

}

int main() {
    string expression = "2+3*4*5-2*5*9";
    BinaryTree* root = expressionToBinaryTree(expression);

    if (evaluateBinaryTree(root)) {
        cout << root -> data << endl;
    }


    return 0;
}
```
